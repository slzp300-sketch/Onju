### 주간 슬롯 잠금 해제 (`src/utils/slots.ts`)

```typescript
import { WeeklyGoal, User } from '../types';

/**
 * 지난 주 달성률을 계산하고 슬롯 잠금 해제 여부를 반환한다.
 * - 달성률 80% 이상 && 현재 슬롯 < 5 → 슬롯 +1
 */
export function checkSlotUnlock(
  lastWeekGoals: WeeklyGoal[],
  currentSlots: number
): { shouldUnlock: boolean; newSlotCount: number; currentRate: number } {
  if (lastWeekGoals.length === 0) {
    return { shouldUnlock: false, newSlotCount: currentSlots, currentRate: 0 };
  }

  const avgRate =
    lastWeekGoals.reduce((sum, g) => sum + g.completionRate, 0) /
    lastWeekGoals.length;

  const shouldUnlock = avgRate >= 80 && currentSlots < 5;

  return {
    shouldUnlock,
    newSlotCount: shouldUnlock ? currentSlots + 1 : currentSlots,
    currentRate: Math.round(avgRate),
  };
}

/**
 * 이번 주 사용 가능한 슬롯 수를 반환한다.
 * User.weeklyGoalSlots를 기준으로 사용.
 */
export function getAvailableSlots(user: User, currentWeekGoalCount: number) {
  return {
    total: user.weeklyGoalSlots,
    used: currentWeekGoalCount,
    remaining: user.weeklyGoalSlots - currentWeekGoalCount,
    isMaxed: user.weeklyGoalSlots >= 5,
  };
}
```

### 달성률 계산 (`src/utils/completion.ts`)

```typescript
import { RoutineLog, DailyRoutine } from '../types';
import { startOfWeek, endOfWeek, eachDayOfInterval, format } from 'date-fns';

/**
 * 특정 날짜 범위의 루틴 달성률 계산
 */
export function calcCompletionRate(
  routines: DailyRoutine[],
  logs: RoutineLog[],
  startDate: Date,
  endDate: Date
): number {
  const days = eachDayOfInterval({ start: startDate, end: endDate });
  let total = 0;
  let completed = 0;

  days.forEach((day) => {
    const dateStr = format(day, 'yyyy-MM-dd');
    routines.forEach((routine) => {
      total++;
      const log = logs.find(
        (l) => l.routineId === routine.id && l.date === dateStr
      );
      if (log?.completed) completed++;
    });
  });

  return total === 0 ? 0 : Math.round((completed / total) * 100);
}

/**
 * 오늘의 Track별 달성률
 */
export function getTodayRates(
  personalRoutines: DailyRoutine[],
  faithRoutines: DailyRoutine[],
  logs: RoutineLog[],
  today: string // YYYY-MM-DD
) {
  const getTodayRate = (routines: DailyRoutine[]) => {
    if (routines.length === 0) return 0;
    const done = routines.filter((r) =>
      logs.find((l) => l.routineId === r.id && l.date === today && l.completed)
    ).length;
    return Math.round((done / routines.length) * 100);
  };

  return {
    personal: getTodayRate(personalRoutines),
    faith: getTodayRate(faithRoutines),
  };
}
```

## 주요 Zustand 스토어

### `src/store/routineStore.ts`

```typescript
import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import { DailyRoutine, RoutineLog } from '../types';
import { format } from 'date-fns';

interface RoutineState {
  personalRoutines: DailyRoutine[];
  faithRoutines: DailyRoutine[];
  logs: RoutineLog[];

  // 오늘 루틴 체크/해제
  toggleRoutineLog: (routineId: string, date?: string) => void;

  // 루틴 순서 변경 (@dnd-kit)
  reorderRoutines: (type: 'personal' | 'faith', oldIndex: number, newIndex: number) => void;

  // 오늘 특정 루틴의 완료 여부
  isCompleted: (routineId: string, date?: string) => boolean;
}

export const useRoutineStore = create<RoutineState>()(
  persist(
    (set, get) => ({
      personalRoutines: [],
      faithRoutines: [],
      logs: [],

      toggleRoutineLog: (routineId, date) => {
        const today = date ?? format(new Date(), 'yyyy-MM-dd');
        const { logs } = get();
        const existing = logs.find(
          (l) => l.routineId === routineId && l.date === today
        );

        if (existing) {
          set({
            logs: logs.map((l) =>
              l.routineId === routineId && l.date === today
                ? { ...l, completed: !l.completed }
                : l
            ),
          });
        } else {
          set({
            logs: [
              ...logs,
              {
                id: `log-${Date.now()}`,
                routineId,
                userId: 'user-1',
                date: today,
                completed: true,
                completedAt: new Date().toISOString(),
              },
            ],
          });
        }
      },

      reorderRoutines: (type, oldIndex, newIndex) => {
        const key = type === 'personal' ? 'personalRoutines' : 'faithRoutines';
        const routines = [...get()[key]];
        const [moved] = routines.splice(oldIndex, 1);
        routines.splice(newIndex, 0, moved);
        set({ [key]: routines.map((r, i) => ({ ...r, order: i })) });
      },

      isCompleted: (routineId, date) => {
        const today = date ?? format(new Date(), 'yyyy-MM-dd');
        const log = get().logs.find(
          (l) => l.routineId === routineId && l.date === today
        );
        return log?.completed ?? false;
      },
    }),
    { name: 'routine-store' }
  )
);
```

**표시 요소:**
- 현재 슬롯 현황: `SlotBadge` (예: "3/3 사용 중")
- 지난 주 달성률 + 슬롯 잠금 해제 가능 여부 안내
  - 달성률 >= 80%: "🎉 다음 주 슬롯 +1 해금 예정" 배너 표시
  - 달성률 < 80%: "지난 주 달성률 XX% — 80% 이상 달성 시 슬롯 확장" 안내
- 이번 주 목표 카드 목록 (completionRate 프로그레스 바 포함)
- 새 목표 추가 버튼 (슬롯 초과 시 비활성화 + 안내 메시지)

### `/groups` — 소모임 피드 (`Groups.tsx`)

**표시 요소:**
- 탭: 전체 | 모집 중 | 진행 중
- 그룹 카드: 제목, 목표, 기간, 현재/최대 인원, 상태 뱃지
- 무한 스크롤 (TanStack Query `useInfiniteQuery`)
- 상단 "소모임 만들기" 버튼

### `/groups/:id` — 소모임 룸 (`GroupDetail.tsx`)

**표시 요소:**
- 그룹 헤더: 제목, 목표, 기간, 멤버 수
- 내 공유 설정: 어떤 목표/루틴을 공유할지 토글
- 멤버 피드: `MemberProgressCard` 목록
  - 각 멤버의 오늘 Track A/B 달성률
  - 이번 주 누적 달성률
  - 연속 달성 스트릭(streak)
- 모집 중 상태: 참여 신청 버튼 + 현재 인원 표시
