# 직장생활조 — 일요일 주간 리뷰 기능 스펙 v3

> 기존 스펙(v1, v2)에 추가되는 일요일 리뷰 & 루틴 개선 기능

---

## 개념 정리

| 구분 | 내용 |
|------|------|
| 한 주 기간 | 월요일 00:00 ~ 일요일 23:59 |
| 월~토 | 루틴 실행 + 체크만 |
| 일요일 | 루틴 실행 + 체크 (동일) **+ 주간 리뷰 & 개선 기능 추가** |
| 리뷰 대상 | 해당 주 월요일 ~ 일요일 전체 (일요일 루틴 포함) |
| 다음 주 적용 | 리뷰에서 변경한 루틴/목표는 다음 주 월요일 00:00부터 적용 |

// 주간 리뷰 레코드
export interface WeeklyReview {
  id: string;
  userId: string;
  weekNumber: number;   // ISO week (월요일 기준)
  year: number;

  // 자동 집계 데이터
  personalRate: number;       // 개인 루틴 주간 달성률 (0-100)
  faithRate: number;          // 신앙 루틴 주간 달성률 (0-100)
  goalAchievedCount: number;  // 달성한 주간 목표 수
  goalTotalCount: number;     // 전체 주간 목표 수

  // 사용자 입력 데이터
  mood: 'hard' | 'normal' | 'easy' | null;
  goalRatings: Record<string, 1 | 2 | 3 | 4 | 5>; // goalId → 별점
  comment: string;            // 한 줄 소감 (optional)
  intention: string;          // 다음 주 집중 한 가지

  // 소모임 공유 설정
  shareToGroups: string[];    // 공유할 그룹 ID 목록

  // 루틴 조정 내역
  routineChanges: RoutineChange[];

  completedAt: string | null; // null이면 미완료
  createdAt: string;
}

// 루틴 조정 내역
export interface RoutineChange {
  routineId: string;
  action: 'keep' | 'edit' | 'delete' | 'add';
  // action === 'edit'일 때 변경 내용
  changes?: {
    title?: string;
    frequency?: DailyRoutine['frequency'];
    weeklyGoalId?: string | null;
  };
  // action === 'add'일 때 새 루틴 정의
  newRoutine?: Omit<DailyRoutine, 'id' | 'userId' | 'createdAt'>;
  appliedAt?: string; // 다음 주 월요일 00:00
}

// 소모임 주간 나눔 (리뷰 공유)
export interface GroupWeeklyShare {
  id: string;
  groupId: string;
  userId: string;
  userName: string;
  weekNumber: number;
  year: number;
  personalRate: number;
  faithRate: number;
  comment: string;      // WeeklyReview.comment만 공유
  intention: string;    // WeeklyReview.intention만 공유
  // goalRatings, mood, routineChanges는 비공개
  sharedAt: string;
}

## 날짜 유틸 추가 (`src/utils/date.ts`)

```typescript
import {
  startOfWeek, endOfWeek, getDay, getISOWeek, getYear,
  format, isWithinInterval, addWeeks
} from 'date-fns';
import { ko } from 'date-fns/locale';

// 한 주 = 월요일 시작
const WEEK_OPTIONS = { weekStartsOn: 1 } as const;

// 오늘이 일요일인지 확인
export const isSunday = (date: Date = new Date()): boolean =>
  getDay(date) === 0;

// 이번 주 월~일 범위
export const getCurrentWeekRange = (date: Date = new Date()) => ({
  start: startOfWeek(date, WEEK_OPTIONS),
  end: endOfWeek(date, WEEK_OPTIONS),
});

// 주간 리뷰 완료 여부 확인
export const isReviewCompleted = (
  reviews: WeeklyReview[],
  weekNumber: number,
  year: number
): boolean =>
  reviews.some(
    r => r.weekNumber === weekNumber && r.year === year && r.completedAt !== null
  );

// 다음 주 첫날 (월요일)
export const getNextWeekMonday = (date: Date = new Date()): Date => {
  const monday = startOfWeek(date, WEEK_OPTIONS);
  return addWeeks(monday, 1);
};

// 루틴 변경 적용 시각 (다음 주 월요일 00:00)
export const getNextWeekApplyTime = (): string =>
  format(getNextWeekMonday(), "yyyy-MM-dd'T'00:00:00");

// 주간 표시 텍스트 (예: "5월 19일(월) ~ 5월 25일(일)")
export const getWeekRangeText = (date: Date = new Date()): string => {
  const { start, end } = getCurrentWeekRange(date);
  return `${format(start, 'M월 d일(EEE)', { locale: ko })} ~ ${format(end, 'M월 d일(EEE)', { locale: ko })}`;
};
```

## 시스템 추천 로직 (`src/utils/routineRecommendation.ts`)

```typescript
import { DailyRoutine, RoutineLog, WeeklyReview } from '../types';
import { calcCompletionRate } from './completion';
import { getCurrentWeekRange } from './date';

export type RecommendationType =
  | 'reduce_frequency'   // 빈도 줄이기
  | 'change_time'        // 시간대 변경
  | 'take_break'         // 잠시 쉬기
  | 'increase_challenge' // 난이도 높이기
  | 'keep_going';        // 그대로 유지

export interface RoutineRecommendation {
  routineId: string;
  type: RecommendationType;
  message: string;
}

export function getRoutineRecommendations(
  routines: DailyRoutine[],
  logs: RoutineLog[],
  pastReviews: WeeklyReview[] // 최근 3주
): RoutineRecommendation[] {
  const { start, end } = getCurrentWeekRange();

  return routines.map(routine => {
    const thisWeekRate = calcCompletionRate([routine], logs, start, end);

    // 최근 2주 달성률 확인
    const recentRates = pastReviews.slice(0, 2).map(review => {
      // review 기간의 달성률 추출 (review에 저장된 데이터 활용)
      return review.personalRate; // 간략화; 실제로는 루틴별 저장 필요
    });

    const consecutiveLow = recentRates.every(r => r < 50) && thisWeekRate < 50;
    const consecutiveHigh = recentRates.every(r => r === 100) && thisWeekRate === 100;

    if (consecutiveLow) {
      return {
        routineId: routine.id,
        type: 'take_break',
        message: '2주 연속 달성률이 낮아요. 잠시 쉬는 것도 방법이에요.',
      };
    }
    if (thisWeekRate < 50) {
      return {
        routineId: routine.id,
        type: 'change_time',
        message: '시간대를 바꿔보는 건 어떨까요?',
      };
    }
    if (thisWeekRate < 70) {
      return {
        routineId: routine.id,
        type: 'reduce_frequency',
        message: '빈도를 줄여서 꾸준히 해보세요.',
      };
    }
    if (consecutiveHigh) {
      return {
        routineId: routine.id,
        type: 'increase_challenge',
        message: '3주 연속 완벽해요! 난이도를 높여볼까요?',
      };
    }
    return {
      routineId: routine.id,
      type: 'keep_going',
      message: '잘 하고 있어요. 이대로 유지해요.',
    };
  });
}
```

## 루틴 변경 적용 로직

```typescript
// src/utils/applyRoutineChanges.ts

export async function applyNextWeekChanges(
  changes: RoutineChange[],
  routineStore: RoutineStore
) {
  const nextMonday = getNextWeekMonday();

  for (const change of changes) {
    switch (change.action) {
      case 'delete':
        // isActive = false (소프트 딜리트, 히스토리 보존)
        await routineStore.deactivateRoutine(change.routineId);
        break;

      case 'edit':
        // 기존 루틴 업데이트
        await routineStore.updateRoutine(change.routineId, change.changes!);
        break;

      case 'add':
        // 새 루틴 생성
        await routineStore.addRoutine({
          ...change.newRoutine!,
          createdAt: nextMonday.toISOString(),
        });
        break;

      case 'keep':
      default:
        // 변경 없음
        break;
    }
  }
}

// 앱 진입 시 (매주 월요일 0시 이후) 자동 실행
// useEffect에서 lastAppliedWeek vs currentWeek 비교 후 처리
```
